Python FaceDetection

   Face Detection คืออะไร คำตอบง่ายๆคือการตรวจจับใบหน้าเมี๊ยว
แล้วทำอะไรได้บ้าง ก็สามารถเอาไปทำเครื่องตรวจจับอารมณ์ ระบบจดจำใบหน้า(Face Recognition) ได้นั่นเอง

มาเริ่มกันเลย!

ก่อนอื่นไปที่ cmd แล้วพิมพ์ pip install opencv
แล้วโหลดไฟล์haarcascade_frontalface_alt2.xml
พอติดตั้งเสร็จแล้วให้ไปที่pythonแล้วพิมพ์โค้ดตามเลย

สำหรับรูป


import cv2
   #import opencv
face_cascade = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
   # โหลด cascade (ไฟล์ตรวจจับใบหน้า)
img = cv2.imread('ไฟล์รูป.นามสกุลไฟล์')
   #อ่านรูป
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
   #เปลี่ยนรูปเป็นสีเท่า(ง่ายต่อการคำนวน)
faces = face_cascade.detectMultiScale(gray, 1.1, 4)
   #ตรวจจับหน้า
for (x, y, w, h) in faces:
   #เพิ่มกรอปสี่เหลี่ยมบริเวณที่เจอหน้า
   cv2.rectangle(img, (x, y), (x+w, y+h), (255, 0, 0), 2)
   cv2.imshow('img', img)
   #โชว์รูป
cv2.waitKey()

import cv2
   #import opencv
face_cascade = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
   # โหลด cascade (ไฟล์ตรวจจับใบหน้า)
cap = cv2.VideoCapture(0)
   # เลือกกล้องบางเครื่องอาจเป็น 1
# cap = cv2.VideoCapture('filename.mp4')
   # สามารถนำเข้าเป็นvideoได้
while True:
   _, img = cap.read()
      # อ่านรูปทีละเฟรม
   gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
   # เปลี่ยนรูปเป็นgrayscale
   faces = face_cascade.detectMultiScale(gray, 1.1, 4)
   # จับใบหน้า
      for (x, y, w, h) in faces:       cv2.rectangle(img, (x, y), (x+w, y+h), (255, 0, 0), 2)       cv2.imshow('img', img)       k = cv2.waitKey(30) & 0xff    # หยุดเมื่อคีย์บอร์ดถูกกด
   if k==27:
      break cap.release()
